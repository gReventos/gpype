    def __init__(self,
                 serial: str = None,
                 channel_count: int = None,
                 **kwargs):

        if channel_count is None:
            channel_count = MAX_NUM_CHANNELS
        channel_count = max(1, min(channel_count, MAX_NUM_CHANNELS))
        self._target_sn = serial

        # Instantiate amplifier to get sampling_rate for super() call.
        # Note: super().__init__ must be called before self._device is
        # assigned, as the base class __init__ sets self._device to None.
        temp_device_for_sr = ble.Amplifier(serial=serial)
        super().__init__(channel_count=channel_count,
                         sampling_rate=temp_device_for_sr.sampling_rate,
                         **kwargs)

        # Assign the amplifier instance to self._device and finish config.
        self._device = temp_device_for_sr
        self._device.set_data_callback(self._data_callback)

    def step(self, data: dict[str, np.ndarray]) -> dict[str, np.ndarray]:  # noqa: E501
        return {PORT_OUT: data[PORT_IN]}



    def _data_callback(self, sender, data: np.ndarray):
        # sender argument is ignored but present to match gtec_ble's call.
        cc_key = AmplifierSource.Configuration.Keys.CHANNEL_COUNT
        self.cycle(data={Constants.Defaults.PORT_IN:
                         data.copy()[np.newaxis, :self.config[cc_key][0]]})
