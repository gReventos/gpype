class SineGenerator(FixedRateSource):
    """
    Generates sinusoidal signals with configurable parameters.

    This source generates continuous sine wave signals at a specified sampling
    rate. It's commonly used for testing signal processing pipelines, creating
    reference signals, or simulating oscillatory phenomena. The generator
    maintains phase continuity across processing blocks.

    Parameters
    ----------
    sampling_rate : float
        Sampling rate for signal generation in Hz.
    channel_count : int
        Number of output channels to generate.
    frequency : float, optional
        Frequency of the sine wave in Hz. Must be positive. Defaults to 1.0.
    amplitude : float, optional
        Peak amplitude of the sine wave. Must be positive. Defaults to 1.0.
    phase : float, optional
        Initial phase offset in radians. Defaults to 0.0.

    Examples
    --------
    >>> # Generate 10 Hz sine wave at 250 Hz sampling rate
    >>> sine_gen = SineGenerator(
    ...     sampling_rate=250.0,
    ...     channel_count=1,
    ...     frequency=10.0,
    ...     amplitude=1.0
    ... )

    >>> # Generate multi-channel test signal with phase offset
    >>> test_signal = SineGenerator(
    ...     sampling_rate=1000.0,
    ...     channel_count=8,
    ...     frequency=50.0,
    ...     amplitude=0.5,
    ...     phase=np.pi/4
    ... )

    Notes
    -----
    - Phase continuity is maintained between consecutive processing blocks
    - Frequency should be much less than sampling_rate/2 (Nyquist frequency)
    - All channels generate the same sine wave signal
    """

    def step(self, data: dict[str, np.ndarray]) -> dict[str, np.ndarray]:
        # Calculate time step
        dt = 1.0 / self.config[SineGenerator.Configuration.Keys.SAMPLING_RATE]

        # Generate sine wave
        freq = self.config[SineGenerator.Configuration.Keys.FREQUENCY]
        amp = self.config[SineGenerator.Configuration.Keys.AMPLITUDE]
        phase = self.config[SineGenerator.Configuration.Keys.PHASE]
        t = np.linspace(self._time, self._time + dt, num=1, endpoint=False)
        sine_wave = amp * np.sin(2 * np.pi * freq * t + phase)

        self._time += dt * len(t)

        # Create output array with correct channel
        # count and broadcast sine wave to all channels
        channel_count = self.config[SineGenerator.Configuration.Keys.CHANNEL_COUNT][0]  # noqa: E501
        output = np.zeros((1, channel_count))
        for i in range(channel_count):
            output[:, i] = sine_wave  # Copy sine wave to each channel

        return {OUT_PORT: output}
