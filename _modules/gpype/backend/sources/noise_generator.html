class NoiseGenerator(FixedRateSource):
    """
    Generates white Gaussian noise with configurable variance.

    This source generates continuous white Gaussian noise signals at a
    specified sampling rate. It's commonly used for testing system
    robustness, simulating measurement noise, or creating noise baselines
    for signal-to-noise ratio analysis. Each sample is independently
    drawn from a normal distribution.

    Parameters
    ----------
    sampling_rate : float
        Sampling rate for noise generation in Hz.
    channel_count : int
        Number of output channels to generate.
    variance : float, optional
        Variance of the Gaussian noise. Must be positive. The standard
        deviation will be sqrt(variance). Defaults to 1.0.

    Examples
    --------
    >>> # Generate white noise with unit variance
    >>> noise_gen = NoiseGenerator(
    ...     sampling_rate=250.0,
    ...     channel_count=1,
    ...     variance=1.0
    ... )

    >>> # Generate low-amplitude noise for 8 channels
    >>> background_noise = NoiseGenerator(
    ...     sampling_rate=1000.0,
    ...     channel_count=8,
    ...     variance=0.01  # std = 0.1
    ... )

    Notes
    -----
    - Each channel generates independent noise
    - Noise values follow N(0, variance) distribution
    - Uses numpy's default random number generator
    - Output data type matches the system's configured data type
    """

    def __init__(self,
                 sampling_rate: float,
                 channel_count: int,
                 variance: float = 1,
                 **kwargs):
        if variance <= 0:
            raise ValueError("variance must be greater than zero.")
        output_ports = kwargs.pop(self.Configuration.Keys.OUTPUT_PORTS,
                                  [OPort.Configuration()])
        FixedRateSource.__init__(self,
                                 sampling_rate=sampling_rate,
                                 channel_count=channel_count,
                                 variance=variance,
                                 output_ports=output_ports,
                                 **kwargs)

    def step(self, data: dict[str, np.ndarray]) -> dict[str, np.ndarray]:  # noqa: E501
        dims = (1, self.config[self.Configuration.Keys.CHANNEL_COUNT][0])
        std = Constants.DATA_TYPE(np.sqrt(self.config[self.Configuration.Keys.VARIANCE]))  # noqa: E501
        rng = np.random.default_rng()
        val = rng.standard_normal(size=dims,
                                  dtype=Constants.DATA_TYPE) * std
        return {OUT_PORT: val}
