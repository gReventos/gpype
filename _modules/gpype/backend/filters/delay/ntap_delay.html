class NTapDelay(IONode):
    """
    Introduces a configurable N-sample delay to the input signal.

    The NTapDelay filter delays the input signal by N samples, where N is
    the number of 'taps'. This is commonly used in signal processing for
    creating delayed versions of signals, implementing FIR filters, or
    synchronization purposes. The filter maintains an internal buffer to
    store previous samples.

    Parameters
    ----------
    taps : int
        Number of samples to delay the signal. Must be non-negative.
        - taps = 0: No delay (immediate passthrough)
        - taps = 1: Delay by 1 sample
        - taps = N: Delay by N samples

    Examples
    --------
    >>> # Delay signal by 5 samples
    >>> delay = NTapDelay(taps=5)

    >>> # No delay (passthrough)
    >>> passthrough = NTapDelay(taps=0)

    Notes
    -----
    - During the initial N samples, the output will be zeros until the
      buffer is filled
    - The filter requires N+1 samples in its internal buffer before
      producing delayed output
    - Memory usage scales with the number of taps and channels
    """

            Additional keyword arguments passed to the superclass.

        if taps < 0:
            raise ValueError("Number of taps must be non-negative.")
        super().__init__(taps=taps, **kwargs)
        self._buffer = None

    def step(self,
             data: dict[str, np.ndarray]) -> dict[str, np.ndarray]:  # noqa: E501
        # Get input data and number of taps
        data_in = data[PORT_IN]
        taps = self.config[NTapDelay.Configuration.Keys.TAPS]

        # Special case: zero taps means immediate output
        if taps == 0:
            return {PORT_OUT: data_in}

        # Add new sample to buffer
        self._buffer.append(data_in)

        # For N taps, we need N+1 samples in buffer
        # before outputting delayed samples
        # (N samples for delay + 1 current sample)
        if len(self._buffer) < taps + 1:
            return {PORT_OUT: np.zeros_like(data_in)}

        # Keep only the samples we need (N+1)
        while len(self._buffer) > taps + 1:
            self._buffer.pop(0)

        # Return the oldest sample (delayed by N samples)
        return {PORT_OUT: self._buffer[0]}
