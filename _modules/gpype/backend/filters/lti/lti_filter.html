class LTIFilter(IONode):
    """
    Base class for Linear Time-Invariant (LTI) digital filters.

    This class implements a generic LTI filter using transfer function
    coefficients (numerator 'b' and denominator 'a' polynomials). It uses
    scipy's lfilter for efficient filtering and maintains internal state
    for continuous processing of streaming data.

    Parameters
    ----------
    b : np.ndarray
        Numerator coefficients of the transfer function. Must not be empty.
    a : np.ndarray
        Denominator coefficients of the transfer function. Must not be empty
        and the first element must be non-zero.

    Notes
    -----
    The transfer function is defined as:

    H(z) = (b[0] + b[1]*z^-1 + ... + b[M]*z^-M) /
           (a[0] + a[1]*z^-1 + ... + a[N]*z^-N)

    This is a base class - use specific filter implementations like
    BandPassFilter, LowPassFilter, etc. for common filter types.

    Examples
    --------
    >>> # Simple moving average filter (FIR)
    >>> b = np.ones(5) / 5  # Average of 5 samples
    >>> a = np.array([1])   # No feedback
    >>> ma_filter = LTIFilter(b=b, a=a)
    """

    def __init__(self,

        # Validate the coefficients

            raise ValueError("Filter coefficients 'b' and 'a' must not be "

        super().__init__(b=b, a=a, **kwargs)

    def setup(self,
              data: dict[str, np.ndarray],
              port_metadata_in: dict[str, dict]) -> dict[str, dict]:

        # Retrieve metadata and initialize the filter state
        md = port_metadata_in[PORT_IN]
        channel_count = md.get(Constants.Keys.CHANNEL_COUNT)
        if channel_count is None:
            raise ValueError("Channel count must be provided in metadata.")

        b = self.config[LTIFilter.Configuration.Keys.B]
        a = self.config[LTIFilter.Configuration.Keys.A]

        # Ensure the coefficients are valid
        if len(a) < 1 or a[0] == 0:
            raise ValueError("Invalid 'a' coefficients: first element must be "
                             "non-zero.")

        # Initialize filter state for each channel
        zi = lfilter_zi(b, a)  # Initial condition for the filter
        self._z = np.tile(zi, (channel_count, 1)).T

        return super().setup(data, port_metadata_in)

    def step(self,
             data: dict[str, np.ndarray]) -> dict[str, np.ndarray]:  # noqa: E501
        data_in = data[PORT_IN]

        # Retrieve coefficients
        b = self.config[LTIFilter.Configuration.Keys.B]
        a = self.config[LTIFilter.Configuration.Keys.A]

        # Ensure input data is compatible
        if data_in.ndim != 2:
            raise ValueError("Input data must be a 2D array (samples x "
                             "channels).")

        # Apply the filter to each channel
        data_out, self._z = lfilter(b, a, data_in, axis=0, zi=self._z)

        return {PORT_OUT: data_out}
