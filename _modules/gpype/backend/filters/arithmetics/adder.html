    def __init__(self,
                 constant: float = 0.0,
                 operation: 'Operation | str' = Operation.ADD,
                 **kwargs):
        """
        Initialize the AddFilter with the given parameters.

        Parameters
        ----------
        constant : float, optional
            Constant to add or subtract. Defaults to 0.0.
        operation : {'ADD' or 'SUBSTRACT'}, optional
            Operation type. Defaults to 'add'.
        **kwargs
            Additional keyword arguments passed to the superclass.
        """
        try:
            # Validate the operation string by
            # trying to convert it to the Enum
            _ = Adder.Operation(operation)
        except ValueError:
            valid_ops = [e.value for e in Adder.Operation]
            raise ValueError(f"Operation must be one of {valid_ops}.")
        super().__init__(constant=constant, operation=operation, **kwargs)

    def step(self,
             data: dict[str, np.ndarray]) -> dict[str, np.ndarray]:  # noqa: E501
        data_in = data[PORT_IN]
        constant = self.config[Adder.Configuration.Keys.CONSTANT]
        operation = self.config[Adder.Configuration.Keys.OPERATION]

        if operation == Adder.Operation.ADD:
            data_out = data_in + constant
        elif operation == Adder.Operation.SUBSTRACT:
            data_out = data_in - constant

        return {PORT_OUT: data_out}
