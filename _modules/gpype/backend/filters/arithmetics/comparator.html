    class Operation(str, Enum):
        """
        Enumeration of supported comparison operations.
        """
        GreatEq = "GreaterOrEqualThan"
        Great = "GreaterThan"
        Eq = "EqualThan"
        LowEq = "LowerOrEqualThan"
        Low = "LowerThan"

    def __init__(self,
                 operation: 'Operation | str' = Operation.GreatEq,
                 **kwargs):
        """
        Initialize the Comparator with a specific comparison operation.

        Parameters
        ----------
        operation : Operation or str
            The comparison operation to perform (either enum member or
            string value).
        """
        # Validate and convert operation if it's a string
        try:
            if isinstance(operation, str):
                operation = Comparator.Operation(operation)
        except ValueError:
            # Raise a more informative error if string conversion fails
            valid_ops = [op.value for op in Comparator.Operation]
            raise ValueError(
                f"Invalid operation string '{operation}'. "
                f"Must be one of {valid_ops}."
            )

        # Ensure it's a valid enum member after potential conversion
        if not isinstance(operation, Comparator.Operation):
            # This case handles non-string, non-enum inputs
            raise TypeError(
                f"Operation must be a Comparator.Operation enum member "
                f"or a valid string, not {type(operation).__name__}."
            )

        super().__init__(
            operation=operation,
            input_ports=[
                IPort.Configuration(name=self.PORT_IN_1),
                IPort.Configuration(name=self.PORT_IN_2)
            ],
            **kwargs
        )

    def setup(self,
              data: dict[str, np.ndarray],
              port_metadata_in: dict[str, dict]) -> dict[str, dict]:
        sr_key = "sampling_rate"
        sampling_rates = [md.get(sr_key, None) for md in port_metadata_in.values()]  # noqa: E501
        sampling_rates = [sr for sr in sampling_rates if sr is not None]

        if len(set(sampling_rates)) != 1:
            raise ValueError("All ports must have the same sampling rate.")
        sr = sampling_rates[0]

        port_metadata_out: dict[str, dict] = {}
        cc_key = "channel_count"
        cc = 1
        op_key = self.Configuration.Keys.OUTPUT_PORTS
        name_key = OPort.Configuration.Keys.NAME
        for op in self.config[op_key]:
            metadata = {}
            metadata[cc_key] = cc
            metadata[sr_key] = sr
            port_metadata_out[op[name_key]] = metadata
        return port_metadata_out

    def step(self,
             data: dict[str, np.ndarray]) -> dict[str, np.ndarray]:  # noqa: E501
        # Get the input signals
        signal1 = data[self.PORT_IN_1]
        signal2 = data[self.PORT_IN_2]

        # Perform the comparison based on the specified operation
        operation = self.config[Comparator.Configuration.Keys.OPERATION]
        if operation == self.Operation.GreatEq:
            data_out = signal1 >= signal2
        elif operation == self.Operation.Great:
            data_out = signal1 > signal2
        elif operation == self.Operation.Eq:
            data_out = signal1 == signal2
        elif operation == self.Operation.LowEq:
            data_out = signal1 <= signal2
        elif operation == self.Operation.Low:
            data_out = signal1 < signal2

        # Return the result
        return {PORT_OUT: data_out.astype(Constants.DATA_TYPE)}
