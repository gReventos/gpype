class Multiplier(IONode):
    """
    Multiplies the input signal by a constant factor.

    The Multiplier filter applies scalar multiplication to the input signal,
    where each sample is multiplied by a configurable constant value. This
    is useful for scaling, amplification, or attenuation of signals.

    Parameters
    ----------
    constant : float, optional
        The multiplication factor to apply to the input signal.
        Defaults to 1.0 (no change).

    Examples
    --------
    >>> # Scale signal by factor of 2.5
    >>> multiplier = Multiplier(constant=2.5)
    
    >>> # Attenuate signal to half amplitude
    >>> attenuator = Multiplier(constant=0.5)
    """

    def step(self,
             data: dict[str, np.ndarray]) -> dict[str, np.ndarray]:  # noqa: E501
        data_in = data[PORT_IN]
        constant = self.config[Multiplier.Configuration.Keys.CONSTANT]
        data_out = data_in * constant
        return {PORT_OUT: data_out}
