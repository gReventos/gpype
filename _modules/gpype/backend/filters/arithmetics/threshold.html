class Threshold(IONode):
    """
    Binary threshold filter that outputs 1.0 or 0.0 based on comparison.

    The Threshold filter compares each sample of the input signal against a
    configurable threshold value. When the input exceeds the threshold, the
    output is 1.0, otherwise it's 0.0. This creates a binary signal useful
    for event detection, signal conditioning, or creating trigger signals.

    Parameters
    ----------
    threshold : float, optional
        The threshold value for comparison. Input values greater than this
        threshold will produce output of 1.0, while values less than or
        equal will produce 0.0. Defaults to 0.0.

    Examples
    --------
    >>> # Detect positive values
    >>> detector = Threshold(threshold=0.0)

    >>> # Detect significant amplitude (e.g., > 50ÂµV)
    >>> amplitude_detector = Threshold(threshold=50e-6)

    Notes
    -----
    The comparison is strictly greater than (>) the threshold value.
    Output values are cast to the system's configured data type.
    """

    def step(self,
             data: dict[str, np.ndarray]) -> dict[str, np.ndarray]:  # noqa: E501
        data_in = data[PORT_IN]
        threshold = self.config[Threshold.Configuration.Keys.THRESHOLD]
        data_out = (data_in > threshold).astype(Constants.DATA_TYPE)
        return {PORT_OUT: data_out}
