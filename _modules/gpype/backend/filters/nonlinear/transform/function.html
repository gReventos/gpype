    def __init__(self,
                 transform_fn: Callable[[np.ndarray],
                                        np.ndarray],
                 **kwargs):
        # Validate the transform function
        test_input = np.array([1.0, 2.0, 3.0])
        try:
            output = transform_fn(test_input)
            if not isinstance(output, np.ndarray):
                raise TypeError("Transform function must return "
                                + str(np.ndarray) + " array")
        except Exception as e:
            raise ValueError(f"Invalid transform function: {str(e)}")
        super().__init__(transform_fn=transform_fn, **kwargs)

    def step(self,
             data: dict[str, np.ndarray]) -> dict[str, np.ndarray]:  # noqa: E501
        data_in = data[PORT_IN]
        transform_fn = self.config[Function.Configuration.Keys.TRANSFORM_FN]
        data_out = transform_fn(data_in)
        return {PORT_OUT: data_out}
