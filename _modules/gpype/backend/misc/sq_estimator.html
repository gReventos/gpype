    def setup(self,
              data: dict[str, np.ndarray],
              port_metadata_in: dict[str, dict]) -> dict[str, dict]:

        # Retrieve metadata and initialize the filter state
        md = port_metadata_in[PORT_IN]
        channel_count = md.get(Constants.Keys.CHANNEL_COUNT)
        if channel_count is None:
            raise ValueError("Channel count must be provided in metadata.")

        sampling_rate = md.get(Constants.Keys.SAMPLING_RATE)
        if sampling_rate is None:
            raise ValueError("Sampling rate must be provided in metadata.")  # noqa: E501

        from scipy.signal import butter
        order = 2
        f_lo1 = 0.5
        f_lo2 = 10
        f_hi = 15
        self._b_low_bp, self._a_low_bp = butter(
            order,
            np.array([f_lo1, f_lo2]) / (sampling_rate / 2),
            btype='bandpass'
        )

        # Low-pass filter
        self._b_hp, self._a_hp = butter(
            order,
            f_hi / sampling_rate * 2,
            btype='highpass'
        )

        # B = np.uint(np.round(sampling_rate / 2))
        # self._b_buf_hp = [1 / B] * B
        # self._a_buf_hp = [1]
        # self._b_buf_lp = [1 / B] * B
        # self._a_buf_lp = [1]

        alpha = 0.985
        self._b_buf_hp = [1 - alpha]
        self._a_buf_hp = [1, -alpha]
        self._b_buf_low_bp = [1 - alpha]
        self._a_buf_low_bp = [1, -alpha]

        # a = self.config[LTIFilter.Configuration.Keys.A]

        # Ensure the coefficients are valid
        # if len(a) < 1 or a[0] == 0:
        #     raise ValueError("Invalid 'a' coefficients: first "
        #                      "element must be non-zero.")

        # Initialize filter state for each channel
        zi_buf_hp = lfilter_zi(  # Initial condition for the filter
            self._b_buf_hp,
            self._a_buf_hp
        )
        self._z_buf_hp = np.tile(
            zi_buf_hp,
            (channel_count, 1)
        ).T
        zi_buf_low_bp = lfilter_zi(  # Initial condition for the filter
            self._b_buf_low_bp,
            self._a_buf_low_bp
        )
        self._z_buf_low_bp = np.tile(
            zi_buf_low_bp,
            (channel_count, 1)
        ).T
        zi_hp = lfilter_zi(  # Initial condition for the filter
            self._b_hp,
            self._a_hp
        )
        self._z_hp = np.tile(
            zi_hp,
            (channel_count, 1)
        ).T
        zi_low_bp = lfilter_zi(  # Initial condition for the filter
            self._b_low_bp,
            self._a_low_bp
        )
        self._z_low_bp = np.tile(
            zi_low_bp,
            (channel_count, 1)
        ).T

        return super().setup(data, port_metadata_in)

    def step(self,
             data: dict[str, np.ndarray]) -> dict[str, np.ndarray]:  # noqa: E501
        data_in = data[PORT_IN]

        # Ensure input data is compatible
        if data_in.ndim != 2:
            raise ValueError("Input data must be a 2D array (channels x "
                             "samples).")

        # Helper function to process each signal path
        def _process_path(b, a, z, b_buf, a_buf, z_buf, data, target, margin):
            """
            Process a single signal path through filters and transformations.
            """
            # Apply main filter
            x, z_new = lfilter(b, a, data, axis=0, zi=z)
            # Square the output
            x_squared = np.square(x)
            # Apply buffer filter
            x_buf, z_buf_new = lfilter(
                b_buf, a_buf, x_squared, axis=0, zi=z_buf)
            # Compute SQ value
            sq = 10 * np.log10(x_buf + EPS)
            sq = (sq - target) / margin
            sq = abs(sq)
            return sq, z_new, z_buf_new

        # Process high-pass path
        sq_hp, self._z_hp, self._z_buf_hp = _process_path(
            self._b_hp, self._a_hp, self._z_hp,
            self._b_buf_hp, self._a_buf_hp, self._z_buf_hp,
            data_in, self._target_hp, self._margin_hp
        )

        # Process low band-pass path
        sq_low_bp, self._z_low_bp, self._z_buf_low_bp = _process_path(
            self._b_low_bp, self._a_low_bp, self._z_low_bp,
            self._b_buf_low_bp, self._a_buf_low_bp, self._z_buf_low_bp,
            data_in, self._target_low_bp, self._margin_low_bp
        )

        sq = np.maximum(sq_hp, sq_low_bp)

        return {PORT_OUT: sq}
